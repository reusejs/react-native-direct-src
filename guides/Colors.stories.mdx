# Colors
These are all the colors that you can use.

```js
//usage
import { colors } from 'react-native-design-system';

theme.brandColor.primary = color.lime['A700'];
```
<ColorPalette>
  <FlatList
    data={Object.keys(color)}
    keyExtractor={(colorGroup, index) => index.toString()}
    renderItem={({ item: colorGroup }) => (
      <FlatList
        data={Object.keys(color[colorGroup])}
        keyExtractor={hex => hex}
        renderItem={({ item: hex }) => (
          <ColorItem
            title={`colors.${colorGroup}['${hex}']`}
            subtitle={colorGroup.toUpperCase()}
            colors={[color[colorGroup][hex]]}
          />
        )}
      />
    )}
  />
</ColorPalette>
